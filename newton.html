def newton(z0, max_iter=100, acc=1e-10):
    z = z0
    for i in range(max_iter):
        fz = f(z)
        if abs(fz) < acc:
            return i, z
        dz = df(z)
        if dz == 0:
            break
        z = z - fz / dz
    return max_iter, z
